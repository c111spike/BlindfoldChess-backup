const formatForSpeech = (text: string): string => {
  return text
    .split('')
    .map(char => {
      // 1. Check your existing letter map
      if (phoneticMap[char]) return phoneticMap[char];
      
      // 2. Handle Piece abbreviations
      const pieceMap: Record<string, string> = {
        'N': 'Knight', 'B': 'Bishop', 'R': 'Rook', 'Q': 'Queen', 'K': 'King'
      };
      if (pieceMap[char]) return pieceMap[char] + ' ';

      // 3. Handle Actions
      if (char === 'x') return ' takes ';
      
      return char;
    })
    .join('')
    .replace(/\s+/g, ' '); // Clean up extra spaces
};
