Copy to Clipboard (The "Quick" Way)Use the standard Web Clipboard API, which works across all platforms.JavaScriptasync function copyPGN() {
  const pgn = game.pgn(); // From your chess.js instance
  await navigator.clipboard.writeText(pgn);
  toast({ title: "Copied!", description: "PGN copied to clipboard" });
}
2. Native Share (The "Pro" Way)Use @capacitor/share to let users export the file.JavaScriptimport { Share } from '@capacitor/share';

async function sharePGN() {
  const pgn = game.pgn();
  await Share.share({
    title: 'Blindfold Chess Game',
    text: pgn,
    dialogTitle: 'Share your game',
  });
}
