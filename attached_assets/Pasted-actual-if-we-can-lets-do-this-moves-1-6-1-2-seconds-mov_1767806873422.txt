Phase 3: The Time Scramble (Overrides)
Simulates a human panicking as the clock runs out.

Under 60s: All moves = 1 second.

Under 30s: All moves = 0.5 seconds.

const getHumanDelay = (moveNumber: number, pieces: number, timeLeft: number): number => {
  // 1. Check Time Scramble Overrides
  if (timeLeft < 30) return 500;
  if (timeLeft < 60) return 1000;

  const getRandom = (min: number, max: number) => 
    Math.floor(Math.random() * (max - min + 1) + min) * 1000;

  // 2. Phase 1: Early Game
  if (moveNumber <= 6) return getRandom(1, 2);
  if (moveNumber <= 12) return getRandom(2, 3);
  if (moveNumber <= 20) return getRandom(2, 7);

  // 3. Phase 2: Piece-Density Scaling (Move 21+)
  if (pieces >= 26) return getRandom(4, 8);
  if (pieces >= 20) return getRandom(4, 7);
  if (pieces >= 14) return getRandom(3, 6);
  if (pieces >= 9) return getRandom(2, 4);
  return getRandom(1, 2);
};
