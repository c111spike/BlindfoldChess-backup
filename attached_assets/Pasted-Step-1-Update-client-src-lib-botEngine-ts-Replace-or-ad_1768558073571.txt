// client/src/lib/botEngine.ts

export interface BotConfig {
  elo: number;
  depth: number;
  randomPercent: number; // Chance to pick a suboptimal move (0.0 to 1.0)
  skillLevel: number;    // Stockfish native "Skill Level" (0-20)
  description: string;
}

export const getBotConfigFromElo = (elo: number): BotConfig => {
  // Clamp Elo between 400 and 2600
  const safeElo = Math.max(400, Math.min(2600, elo));
  
  let depth = 1;
  let randomPercent = 0;
  let skillLevel = 20; // Default to max skill, reduce for lower Elos
  let description = "";

  if (safeElo < 800) {
    // BEGINNER (400 - 800)
    // High randomness, very low depth.
    // Logic: Linearly reduce randomness from 40% down to 10%
    const progress = (safeElo - 400) / 400; // 0.0 to 1.0
    depth = 1;
    randomPercent = 0.4 - (progress * 0.3); // 0.4 -> 0.1
    skillLevel = 5; 
    description = "Beginner (Makes frequent mistakes)";

  } else if (safeElo < 1200) {
    // CASUAL (800 - 1200)
    // Randomness disappears, Depth increases slightly.
    const progress = (safeElo - 800) / 400;
    depth = 2 + Math.floor(progress * 2); // Depth 2 -> 4
    randomPercent = 0.1 - (progress * 0.1); // 0.1 -> 0.0
    skillLevel = 10;
    description = "Casual (Misses tactics)";

  } else if (safeElo < 1800) {
    // CLUB PLAYER (1200 - 1800)
    // Solid play, increasing depth significantly.
    const progress = (safeElo - 1200) / 600;
    depth = 4 + Math.floor(progress * 6); // Depth 4 -> 10
    randomPercent = 0;
    skillLevel = 20;
    description = "Club Player (Solid play)";

  } else {
    // MASTER (1800 - 2600)
    // High depth, pure calculation.
    const progress = (safeElo - 1800) / 800;
    depth = 10 + Math.floor(progress * 10); // Depth 10 -> 20
    randomPercent = 0;
    skillLevel = 20;
    description = "Master (Precise)";
  }

  return { elo: safeElo, depth, randomPercent, skillLevel, description };
};

import { useState, useEffect } from "react";
import { getBotConfigFromElo } from "../lib/botEngine";

// ... inside your component
const [elo, setElo] = useState<number>(800);
const [botDescription, setBotDescription] = useState<string>("");

// Update description whenever slider moves
useEffect(() => {
  const config = getBotConfigFromElo(elo);
  setBotDescription(config.description);
}, [elo]);

const handleStartGame = () => {
    // Save Elo to local storage so it remembers next time
    localStorage.setItem("preferredElo", elo.toString());
    
    // Get the config to pass to your engine
    const config = getBotConfigFromElo(elo);
    startGame(config);
};

return (
  <div className="p-4 bg-gray-800 rounded-lg">
    <label className="text-white font-bold mb-2 block">
      Opponent Strength: {elo}
    </label>
    
    <input 
      type="range" 
      min="400" 
      max="2600" 
      step="50" // Steps of 50 feel nice and snappy
      value={elo} 
      onChange={(e) => setElo(parseInt(e.target.value))}
      className="w-full h-2 bg-gray-600 rounded-lg appearance-none cursor-pointer accent-blue-500"
    />
    
    <div className="mt-2 text-blue-300 text-sm font-medium text-center">
      {botDescription}
    </div>
  </div>
);
